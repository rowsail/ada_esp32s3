%  Keep a listing's lead-in line -- its caption or commentary (e.g. "fade an
%  LED") -- on the same page as the listing it introduces: forbid a page break
%  between that preceding line and the start of the listing.  lstlisting is built
%  on a trivlist, which sets a NEGATIVE \@beginparpenalty that actively invites a
%  break right before the listing; so we both raise that penalty and \nobreak the
%  junction.  (An earlier \needspace hook reserved room for the listing but issued
%  its forced break BETWEEN the lead-in and the listing -- widowing the very
%  caption we want to keep with it.)  A listing longer than the page still breaks
%  internally; only the lead-in -> first-line junction is protected.
%  Don't let a code listing start at the very bottom of a page and break after
%  only a line or two: require at least this many lines of room, else push the
%  whole block to the next page.  A listing longer than this still breaks (it has
%  to), but never orphans a tiny fragment.
\usepackage{needspace}
\usepackage{etoolbox}
\BeforeBeginEnvironment{lstlisting}{\par\needspace{6\baselineskip}}

%  A listing's lead-in caption / commentary (e.g. "fade an LED").  The key is that
%  the \needspace runs BEFORE the caption is typeset: if the caption plus the
%  first lines of the listing will not fit, the page breaks ABOVE the caption and
%  the two move to the next page together -- so the caption is never widowed onto
%  the page before its listing.  (A bare \needspace AFTER the caption, as the hook
%  above does for un-captioned listings, would instead break between the caption
%  and the code.)  \nopagebreak then glues the caption to the listing.
%  The invisible \hrule contributes a box to the main vertical list, which forces
%  the page builder to run so \pagetotal includes the just-ended paragraph / list
%  BEFORE \needspace measures the room left.  (A bare \penalty does not reliably
%  flush it, so a caption right after \end{description} mis-measured and slipped
%  to the page bottom anyway.)  \needspace then pushes the caption + first listing
%  lines to the next page together if they will not fit; \nopagebreak glues the
%  caption to the listing.
\newcommand{\lstcapcore}[1]{%
  \par\hrule height 0pt\relax
  \needspace{8\baselineskip}\noindent\textbf{#1}\nopagebreak}
%  \lstcap: a plain caption (used for the rare non-example commentary line).
\newcommand{\lstcap}[1]{\lstcapcore{#1}}
%  \lstexample: an auto-numbered worked example.  The number is the section
%  number followed by a per-section example index (e.g. "Example 16.5.1"), so it
%  reads as <chapter>.<section>.<example> and restarts in each section.  Use it as
%  \lstexample{<description>} -- the "Example N:" label is generated, not typed.
\newcounter{example}
\counterwithin{example}{section}
\newcommand{\lstexample}[1]{%
  \refstepcounter{example}%
  \addcontentsline{loe}{example}{\protect\numberline{\theexample}#1}%
  \lstcapcore{Example \theexample: #1}}
